  • 实现
  • tip：new 出来的对象的原型链只能找到Object.prototype，不能找到function的prototype
Object.prototype.addMethod = function(name, fn, isProto = false) {
    if (isProto) {
        this.__proto__[name] = fn;
    } else {
        this[name] = fn;
    }
    return this
}
